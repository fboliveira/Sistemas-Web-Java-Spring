Table events {
  id uuid [primary key]
  description varchar
  type integer [note: "Exemplos: palestra, show, teatro, curso, ..."]
  date datetime
  startSales datetime
  endSales datetime
  price float

  createdAt timestamp
  updatedAt timestamp
}

Table sales {
  id uuid [primary key]
  user_id uuid
  event_id uuid

  purchaseDate datetime
  purchaseStatus integer [note: "Em aberto, Pago, Cancelado, Estornado, ..."]

  createdAt timestamp
  updatedAt timestamp

}

Ref event_sales: events.id > sales.event_id_id
